# 链表

链表是⼀个线性结构

链表同时也是⼀个天然的递归结构。链表结构可以充分利⽤计算机内存空间，实现灵活的内存动态管理。

链表每个元素由一个存储元素本身的节点和一个指向下一个元素的引用(也称指针或链接)组成

链表和数组的区别

  1. 数组使用[]访问元素，链表使用.访问元素
  2. 数组在内存存储的顺序是连续的,链表的元素在内存中不是连续放置的
  3. 链表失去了数组随机读取的优点，查找元素要从头结点开始遍历整个链表,但换来的是增删节点的高性能和便携!
  4. 链表由于增加了结点的指针域，空间开销⽐较⼤
  5. 数组增删元素需要移动元素,只能覆盖原有元素,链表可以直接插入或删除元素,只需要改变指针的指向

// 首先创建节点类
class Node {
    constructor(element){
        this.element = element;
        this.next = null;
    }
}

// 创建链表类
class LinkedList {
    // 初始化链表
    constructor() {
    this.head = null;
    this.count = 0;
}

    // 添加节点
    push(){
        const node = new Node(element);
        let current;
        // 如果链表为空
        if(this.head == null){  
            this.head = node;
        }else {
            current = this.head;

            while(current.next != null) {
                current = current.next;
            }
            //将其next指向新节点
            current.next = node;
        }
        this.count++;
    }

    // 删除节点(根据索引值删除)
    // 删除有2种情况，一种是根据索引值index删除，一种是根据节点值element删除
    // 我们先来实现第一种，第二种方法只需要调用第一种方法即可，这样就只需要维护一种方法！复用代码！
    removeAt(index){
        // 检查越界值
        if(index >= 0 && index < this.count) {
            let current  = this.head;

            // 移除第一项
            if(index === 0) {
                this.head = current.next;
            }else {
                // 重构前的方法
                // let previous;
                // for( let i = 0;i<this.count;i++) {
                //     previous = current;
                //     current = current.next;
                // }
                // // 将previous与current 的下一项链接起来，跳过current 从而移除它
                // previous.next = current.next;

                // 有了根据索引值找方法后的 方法
                const previous = this.getElementAt(index - 1);
                current = previous.next;
                previous.next = current.next;
            }
            this.count--;
            return current.element;
        }
        return undefined;   
    }

    // 获取节点(根据索引值)
    getElementAt(index){
        if(index >= 0 && index < this.count) {
            let current  = this.head;
            for( let i = 0; i < index && current != null; i++){
                current = current.next;
            }
            return current;
        }
        return undefined;
    }

    //插入元素
    insert(element,index){
        // 检查越界值
        if(index >= 0 && index <= this.count) {
            const node = new Node(element);
            if (index === 0){
                // 在第一个位置添加
                const current = this.head;
                node.next = current;
                this.head = node;
                
            } else {
                const previous = this.getElementAt(index - 1);
                const current = previous.next;
                node.next = current;
                previous.next = node;
            }
            this.count++;
            return true;
        }
        return false;
    }

    // 返回一个元素的位置
    indexOf(element) {
        let current = this.head;
        for(let i = 0; i < this.count && current != null; i++) {
            // 为什么不写成 current.element === element呢？而要拆开一个函数呢？
            
            if(this.equalsFn(element,current.element)) {
                return i;
            }
        }
        return -1;
    }

    // 因为当存储复杂对象的时候，我们可能需要比较对象的值，而不是对象本身
    equalsFn(a,b){
        return a === b;
    }

    // 根据值来删除元素
    remove(element){
        const index = this.indexOf(element);
        return this.removeAt(index);
    }

    // 返回链表的长度
    size(){
        return this.count;
    }

    // 判断链表是否为空
    isEmpty(){
        return this.count === 0;
    }

    // 获取头节点
    getHead(){
        return this.head;
    }

    // 清空链表
    clear(){
        this.head = null;
        this.count = 0;
    }

    // 返回链表的字符串表示
    toString(){
        if(this.head == null){
            return '';
        } 
        let objString = `${this.head.element}`;
        let current = this.head.next;
        for( let i = 1; i < this.count; i++) {
            objString = `${objString},${current.element}`;
            current = current.next;
        }
        return objString;
    }

}

判断链表是否有环

代码实现

// 判断链表是否有环

// 1.利用 JSON.stringify() 不能序列化含有循环引用的结构
function hasCycle(head){
  try{
      JSON.stringify(head);
      return false;
  }
  catch(e){
      return true;
  }
}


// 2.快慢指针法 
// 让快指针每次比满指针多走一步，如果相等则有环
function hasCycle(head){
  if(!head || !head.next) {
    return false;
  }
  let fast = head;
  let slow = head;
  while (fast != slow) {
    if(!fast || !fast.next) {
      fast = fast.next.next
      slow = slow.next
    }
  }
  return true;
}

// 3.标记法 
// 给每个已遍历过的节点加标志位，遍历链表，当出现下一个节点已被标志时，则证明单链表有环
function hasCycle(head){
  while(head){
    if(head.flag) {
      return true
    }
    head.flag = true;
    head = head.next;
  }
  return false;
}
