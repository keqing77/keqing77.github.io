# Set

集合

集合，这是一种不允许值重复的无序数据结构，其中的值可以是任意类型

  1. 可以把集合想象成一个既没有重复元素，也没有顺序概念的数组。
  2. 集合有并集 交集 差集 等基本运算

# 实现一个集合Set类

Set在ES6后以及内置在JavaScript中了，但手动实现一个能让你对集合有深刻的了解！

// 本例采用对象来实现Set，你也可以用数组实现
class Set{
  constructor() {
    this.items = {};
  }

  // 检验是否包含某个元素
  has(element) {
    return element in this.items;
  }

  // 添加一个元素到集合中
  add(element) {
    if(!this.has(element)) {
      this.items[element] = element;
      // this.count++;
      return true;
    }
    return false;
  }

  //删除一个元素
  delete(element) {
    if(this.has(element)) {
      delete this.items[element];
      // this.count--;
      return true;
    }
    return false;
  }

  // 清空集合
  clear() {
    this.items = {};
  }
 
  //  返回集合元素个数
  size() {
    return  Object.keys(this.items).length;
  }

  //返回Set 对象中的元素值   
  // 因为在Set中， key和value是相等的，所以keys()方法和values()方法返回的是一样的

  values() {
    return Object.values(this.items);
  }

  keys() {
    return Object.keys(this.items);
  }
}

// 交集
function intersection(otherSet) {
  const intersectionSet = new Set(); //创建1个新的集合 存放交集
  // 遍历this集合，如果otherSet也包含这个值，就把这个值添加到intersectionSet中
  this.values().forEach(value => {
    if(otherSet.has(value)) {
      intersectionSet.add(value);
    }
  })
}

// 差集
function difference(otherSet) {
  const differenceSet = new Set()  //创建1个新的集合,存放差集
  // 遍历this集合，如果otherSet不包含这个值，就把这个值添加到differenceSet中
  this.values().forEach(value => {
    if(!otherSet.has(value)) {
      differenceSet.add(value);
    }
  })
}

// 子集
function  isSubsetOf(otherSet) {
  if(this.size() > otherSet.size()) {
    return false;
  }
  // 遍历this集合，如果otherSet不包含这个值，就返回false
  this.values().forEach(value => {
    if(!otherSet.has(value)) {
      return false;
    }
  })
}
